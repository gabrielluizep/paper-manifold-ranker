\documentclass[tikz, border=5mm]{standalone}
\usepackage{pgfplots}
\usepgfplotslibrary{fillbetween}
\pgfplotsset{compat=1.18}

\begin{document}
\begin{tikzpicture}
    \begin{axis}[
        title={Distributions of $s$, $\mathrm{IQS}_\mathrm{leg}$, and $\mathrm{IQS}_\mathrm{rev}$},
        ylabel={Value},
        xmin=0, xmax=4,
        xtick={1, 2, 3},
        xticklabels={$s$, $\mathrm{IQS}_\mathrm{leg}$, $\mathrm{IQS}_\mathrm{rev}$},
        enlarge y limits=0.05,
        ymajorgrids=true,
    ]

    % --- Position 1: Score ---
    \addplot [name path=score_R, draw=blue!80!black, forget plot] 
        table [x=score_right, y=y, col sep=comma] {violin_coordinates.csv};
    \addplot [name path=score_L, draw=blue!80!black, forget plot]

    \addplot [name path=leg_R, draw=red!80!black, forget plot] 
        table [x=iqs_leg_scaled_right, y=y, col sep=comma] {violin_coordinates.csv};
    \addplot [name path=leg_L, draw=red!80!black, forget plot] 
        table [x=iqs_leg_scaled_left, y=y, col sep=comma] {violin_coordinates.csv};
    \addplot [fill=red!60, fill opacity=0.5] fill between[of=leg_L and leg_R];

    % --- Position 3: IQS Rev Scaled ---
    \addplot [name path=rev_R, draw=green!80!black, forget plot]

    \addplot [name path=rev_L, draw=green!80!black, forget plot] 
        table [x=iqs_rev_scaled_left, y=y, col sep=comma] {violin_coordinates.csv};

    \end{axis}
